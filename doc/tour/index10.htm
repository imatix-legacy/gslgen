This is a simple GSL schema 

.- GSL schema to print calendar as HTML
.output "calendar.htm"
<HTML><BODY>
.for week
<H2>Week starting $(start)</H2>
.  for day
.    if "$(alert?)" <> ""
<FONT COLOR="$(alert)"><EM>
.    endif
<P>$(Name):
.    insert day
.    if "$(alert?)" <> ""
</EM></FONT>
.    endif
.  endfor
<HR>
.endfor
